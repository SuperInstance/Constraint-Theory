Quick Start

// Install
npm install @superinstance/constraint-theory

// Use
import { PythagoreanManifold } from '@superinstance/constraint-theory';

const manifold = new PythagoreanManifold(200);
const vec = [0.6, 0.8];
const [snapped, noise] = manifold.snap(vec);
console.log(snapped); // [0.6, 0.8] - perfect 3-4-5 triangle!
